namespace = spn_xp_sys
#XP system - main event
spn_xp_sys.1000 = {
	hidden = yes
	scope = none

	immediate = {
		every_living_character = {
					limit = {
						OR = {
							has_trait = supernatural_vampire
							has_trait = supernatural_werewolf
							has_trait = supernatural_hybrid
						}
					}
					trigger_event = {
						id = spn_xp_sys.1001
						days = { 1 364 }
					}

		}
	}
}

#XP system - character event
spn_xp_sys.1001 = {
	hidden = yes
	type = character_event

	trigger = {
		OR = {
			AND = {
				has_trait = supernatural_vampire
				has_trait_xp = {
					trait = supernatural_vampire
					track = vampirism
					value < 100
				}
			}
			AND = {
				has_trait = supernatural_werewolf
				has_trait_xp = {
					trait = supernatural_werewolf
					track = lycanthropy
					value < 100
				}
			}
			AND = {
				has_trait = supernatural_hybrid
				has_trait_xp = {
					trait = supernatural_hybrid
					track = hybrid_vampirism
					value < 100
				}
			}
			AND = {
				has_trait = supernatural_hybrid
				has_trait_xp = {
					trait = supernatural_hybrid
					track = hybrid_lycanthropy
					value < 100
				}
			}
		}
	}

	immediate = {
		if = {
			limit = {
				has_trait = supernatural_vampire
			}
			add_trait_xp = {
				trait = supernatural_vampire
				track = vampirism
				value = 1
			}
		}
		if = {
			limit = {
				has_trait = supernatural_werewolf
			}
			add_trait_xp = {
				trait = supernatural_werewolf
				track = lycanthropy
				value = 1
			}
		}
		if = {
			limit = {
				has_trait = supernatural_hybrid
			}
			add_trait_xp = {
				trait = supernatural_hybrid
				track = hybrid_vampirism
				value = 1
			}
		}
		if = {
			limit = {
				has_trait = supernatural_hybrid
			}
			add_trait_xp = {
				trait = supernatural_hybrid
				track = hybrid_lycanthropy
				value = 1
			}
		}
	}
}
