#Embrace Vampirism
spn_embrace_vampirism_decision = {
	ai_check_interval = 0
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	desc = spn_embrace_vampirism_decision_desc
	selection_tooltip = spn_embrace_vampirism_decision_tooltip

	is_shown = {
		has_game_rule = spn_yes_embrace_decision
		spn_has_supernatural_trait = no
	}

	is_valid = {
		spn_has_supernatural_trait = no
	}

	effect = {
		add_character_flag = {
			flag = spn_embrace_vampire_dec
		}

		if = {
			limit = {
				OR = {
					has_game_rule = spn_yes_embrace_events 
					has_game_rule = spn_only_embrace_events
				}
			}
			trigger_event = spn_random_embrace.0002 #Random Vampire Embrace
		}
		else = {
			trigger_event = spn_init.0003 #Init Vampire, with trait selection
		}
		
		show_as_tooltip = {
			custom_tooltip = spn_msg_tooltip_embrace_vampirism_desc
		}
	}
	
	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = 0
	}
}

 spn_satiate_bloodlust_decision = {
 	ai_check_interval = 0
	picture = "gfx/interface/illustrations/decisions/decision_activity.dds"
	desc = spn_satiate_bloodlust_decision_desc
	selection_tooltip = spn_satiate_bloodlust_decision_tooltip
 	cooldown = { years = 10 }
	
 	is_shown = {
 		OR = {
 			has_trait = spn_vampire_1
 			has_trait = spn_vampire_2
 			#has_trait = spn_hybrid_1
			#has_trait = spn_hybrid_2
 		}
 	}
	
 	is_valid = {
 		OR = {
 			has_trait = spn_vampire_1
 			has_trait = spn_vampire_2
 			#has_trait = spn_hybrid_1
			#has_trait = spn_hybrid_2
 		}
		years_from_game_start >= 5
 	}

 	effect = {
		show_as_tooltip = {
			custom_tooltip = spn_msg_tooltip_satiate_bloodlust_desc
		}
		
		stress_impact = { #Stress loss
			base = hunt_stress_loss_value
			spn_vampire_1 = activity_stress_loss_impact
			spn_vampire_2 = activity_stress_loss_impact
			spn_vampire_3 = activity_stress_loss_impact
			#spn_hybrid_1 = activity_stress_loss_impact
			#spn_hybrid_2 = activity_stress_loss_impact
			#spn_hybrid_3 = activity_stress_loss_impact
		}
		
 		trigger_event = {
			id = spn_gain_power.4010 #Determine Which Hunt Event to Trigger
			days = 3
		}
	}
}
