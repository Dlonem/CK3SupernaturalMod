#Randomly Assign traits on startup based on settings
spn_npc_trait_dist_all = {
	trigger = {
		has_game_rule = spn_yes_npc_trait_dist
	}
	effect = {
		every_living_character = {
			limit = {
				spn_meets_dist_criteria = { TIER = tier_empire }
			}
			trigger_event = spn_trait_dist.0001 
		}
	}
}

spn_check_hunter_skill_traits = {
	trigger = {
		has_trait = spn_hunter
	}
	effect = {
		random_list = {
			0 = { #Upgrade Age Trait
				modifier = { #Hunter Age 1
					AND = {
						has_trait = spn_hunter_1
						var:spn_yearly_pulses_since_trait_upgrade >= 5
					}
					add = {
						value = var:spn_yearly_pulses_since_trait_upgrade
						multiply = 15
					}
				}

				modifier = { #Hunter Age 2
					AND = {
						has_trait = spn_hunter_2
						var:spn_yearly_pulses_since_trait_upgrade >= 5
					}
					add = {
						value = var:spn_yearly_pulses_since_trait_upgrade
						multiply = 10
					}
				}

				set_variable = {
					name = spn_yearly_pulses_since_trait_upgrade
					value = 0
				}

				if = { limit = { has_trait = spn_hunter_1 } remove_trait = spn_hunter_1 add_trait = spn_hunter_2 }
				else_if = { limit = { has_trait = spn_hunter_2 } remove_trait = spn_hunter_2 add_trait = spn_hunter_3 }
			}
			100 = { #No Upgrade
				change_variable = {
					name = spn_yearly_pulses_since_trait_upgrade
					add = 1
				}
			}
		}
	}
}
