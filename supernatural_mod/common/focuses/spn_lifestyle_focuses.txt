
vampire_vampirism_focus = {
	lifestyle = vampire_lifestyle

	is_shown = {
		NOT = { government_has_flag = government_is_landless_adventurer }
	}
	is_valid = {
		NOT = { government_has_flag = government_is_landless_adventurer }
	}
	
	modifier = {
		intrigue = 3
	}

	desc = {
		desc = vampire_focus_desc
		desc = line_break
	}

	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_vampire_interest_trigger = yes
			}
			add = 2000
		}
		if = {
			limit = {
				age > 55
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = sadistic
			}
			multiply = 3
		}
	}
	
}

witch_focus = {
	lifestyle = magic_lifestyle
	#icon=witch_focus
	is_valid = {
		OR = {
			has_trait = witch
			any_secret = { secret_type = secret_witch }
		}
		NOT = { has_trait = true_witch }
	 }
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { 
					NOT={
						OR = {
							has_trait = witch
							any_secret = { secret_type = secret_witch }
						}
					}
				 }
				desc = REQUIRMENTS_MAGIC1
				
			}

			triggered_desc = {
				trigger = { 
					
						OR = {
							has_trait = witch
							any_secret = { secret_type = secret_witch }
						}
					
				 }
				 desc = magic_witch_focus_desc
				
			}
		}
		
		
		desc = line_break
	}
	
	modifier = {
		learning = 1
		intrigue = 2
		#monthly_magic_lifestyle_xp_gain_mult = 0.5
	}

	auto_selection_weight = {
		value = 0
		if = {
			limit = {
				OR = {
					has_trait = witch
					any_secret = { secret_type = secret_witch }
				}
			}
			add = 3000
		}
		if = {
			limit = {
				culture = {
					has_cultural_tradition = coven_rule_tradition
				}
			}
			add = 2000
		}
		if = {
			limit = {
				has_trait = source2
			}
			multiply = 2
		}
		if = {
			limit = {
				exists = house
				house = {
					has_house_modifier = witch_coven
				}
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = zealous
			}
			multiply = 0
		}
		if = {
			limit = {
				has_trait = demon
			}
			multiply = 0
		}
		if = {
			limit = {
				has_trait = true_witch
			}
			multiply = 0
		}
	}

}

arcane_focus = {
	lifestyle = magic_lifestyle
	#icon=arcane_focus
	is_valid = {

		OR = {
			has_trait = witch
			has_trait = archmage
			has_trait = true_witch
			any_secret = { secret_type = secret_witch }
			can_unlock_arcane_focus_trigger = yes
		}
		OR = {
			any_character_artifact = { has_variable = magic1 }
			any_character_artifact = { has_variable = magic_ring }
			has_trait = source
			has_trait = source2
			has_perk = magical_experiments_p1_perk
			has_power_place_trigger = yes
		}
	 }
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { 
					NOT={
						AND={
							OR = {
								has_trait = witch

								any_character_artifact = { has_variable = magic1 }

								has_power_place_trigger = yes

						}
					}
				 }
				desc = REQUIRMENTS_MAGIC2

			}

			triggered_desc = {
				trigger = { 
					
						AND={
							OR = {
								has_trait = witch
								has_trait = true_witch
								any_secret = { secret_type = secret_witch }
								}

								has_power_place_trigger = yes
							}
						}
					
				 }
				desc = magic_arcane_focus_desc
				#desc = line_break
			}
		}
		
		desc = line_break
	}
	
	modifier = {
		learning = 2
		intrigue = 1
		#monthly_magic_lifestyle_xp_gain_mult = -0.1
	}

	auto_selection_weight = {
		value = 0

			limit = {
				AND={

						has_trait = witch

						any_character_artifact = { has_variable = magic1 }

						any_character_artifact = { has_variable = singularity2 }
						has_trait = source
						#has_trait = source3
						#has_trait = source4
						has_perk = magical_experiments_p1_perk
						has_character_flag = arcane_unlocked_flag
						has_power_place_trigger = yes
						has_character_modifier = research_cooperation
					}
				}
			}
			add = 22000
		}
		if = {
			limit = {
