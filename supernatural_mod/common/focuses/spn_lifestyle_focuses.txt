
vampire_vampirism_focus = {
	lifestyle = vampire_lifestyle

	is_valid = {
		has_trait = supernatural_vampire
	}
	
	modifier = {
		intrigue = 3
	}

	desc = {
		desc = vampire_focus_desc
		desc = line_break
	}

	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_vampire_interest_trigger = yes
			}
			add = 2000
		}
		if = {
			limit = {
				age > 55
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = sadistic
			}
			multiply = 3
		}
	}
	
	focus_id = 15
}

witch_focus = {
	lifestyle = magic_lifestyle
	#icon=witch_focus
	is_valid = {
		OR = {
			has_trait = witch
			any_secret = { secret_type = secret_witch }
		}
		NOT = { has_trait = true_witch }
	 }
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { 
					NOT={
						OR = {
							has_trait = witch
							any_secret = { secret_type = secret_witch }
						}
					}
				 }
				desc = REQUIRMENTS_MAGIC1
				
			}

			triggered_desc = {
				trigger = { 
					
						OR = {
							has_trait = witch
							any_secret = { secret_type = secret_witch }
						}
					
				 }
				 desc = magic_witch_focus_desc
				
			}
		}
		
		
		desc = line_break
	}
	
	modifier = {
		learning = 1
		intrigue = 2
		#monthly_magic_lifestyle_xp_gain_mult = 0.5
	}

	auto_selection_weight = {
		value = 0
		if = {
			limit = {
				OR = {
					has_trait = witch
					any_secret = { secret_type = secret_witch }
				}
			}
			add = 6500
		}
		if = {
			limit = {
				has_trait = zealous
			}
			multiply = 0
		}
		if = {
			limit = {
				has_trait = demon
			}
			multiply = 0
		}
		if = {
			limit = {
				has_trait = true_witch
			}
			multiply = 0
		}
		if = {
			limit = {
				has_trait = archmage2
			}
			multiply = 0
		}
	}

	focus_id = 20
}

arcane_focus = {
	lifestyle = magic_lifestyle
	#icon=arcane_focus
	is_valid = {
		NOT = {
			has_trait = archmage
		}
		OR = {
			has_trait = witch
			has_trait = true_witch
			any_secret = { secret_type = secret_witch }
		}
		OR = {
			any_character_artifact = { has_variable = magic1 }
			any_character_artifact = { has_variable = magic_ring }
			has_trait = source
			has_perk = magical_experiments_p1_perk
		}
	 }
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					NOT={
						AND={
							OR = {
								has_trait = witch

							}
							OR = {
								any_character_artifact = { has_variable = magic1 }
								any_character_artifact = { has_variable = magic_ring }
								has_trait = source
								has_perk = magical_experiments_p1_perk
							}
						}
					}
				 }
				 
				desc = REQUIRMENTS_MAGIC2
				#desc = line_break
			}

			triggered_desc = {
				trigger = { 
					
						AND={
							OR = {
								has_trait = witch
								has_trait = true_witch
								any_secret = { secret_type = secret_witch }
								}
							OR = {
								any_character_artifact = { has_variable = magic1 }
								any_character_artifact = { has_variable = magic_ring }
								has_trait = source

						}

					}

						any_character_artifact = { has_variable = magic1 }
						any_character_artifact = { has_variable = magic_ring }
						has_trait = source
						has_perk = magical_experiments_p1_perk
					}
				}
			}
			add = 8000
		}
		if = {
			limit = {
				has_trait = true_witch
			}
			add = 8000
		}
		if = {
			limit = {
				has_trait = zealous
			}
			multiply = 0
		}
		if = {
