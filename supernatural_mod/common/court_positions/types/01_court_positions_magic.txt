mage_court_position = {
	skill = learning
	max_available_positions = 1
	category = court_position_category_common

	opinion = {
		value = regular_court_position_opinion
	}
	
	# How high the aptitude has to be to reach the 5 different modifier levels
	aptitude_level_breakpoints = { 20 40 60 80 }

	# How good is this character in this position? root scope is the holder of the court position
	aptitude = {
		value = 1
		add = {
			value = learning
			multiply = 1.2
			max = 29
			desc = court_position_skill_learning
		}
		if = {
			limit = {
				has_trait = true_witch
			}
			add = {
				value = 45
				desc = court_position_witch_trait
			}
		}
		else_if = {
			limit = {
				has_focus = witch_focus
			}
			add = {
				value = 30
				desc = court_position_witch_trait
			}
		}
		else_if = {
			limit = {
				has_trait = witch
			}
			add = {
				value = 20
				desc = court_position_witch_trait
			}
		}
		if = {
			limit = {
				OR = {
					has_trait = archmage
					has_trait = demon
				}
			}
			add = {
				value = 60
				desc = court_position_archmage_trait
			}
		}
		else_if = {
			limit = {
				has_focus = arcane_focus
			}
			add = {
				value = 40
				desc = court_position_arcane_focus
			}
		}
		else_if = {
			limit = {
				is_legendary_mage_trigger = yes
			}
			add = {
				value = 80
				desc = court_position_archmage_trait
			}
		}
		if = {
			limit = {
				OR = {
					has_trait = lifestyle_mystic
				}
			}
			add = {
				value = 10
				desc = court_position_mystic_trait
			}
		}
		add = court_position_aptitude_family_business_value
		add = court_position_aptitude_low_penalty_value
	}

	# Is this an available position for this court? (root is court owner)
	valid_position = { 
		#custom_description = {
		#	text = witchcraft_illegal_custom
		#	NOT = { trait_is_criminal_in_faith_trigger = { TRAIT = witch FAITH = root.faith GENDER_CHARACTER = root } }
		#}
		custom_description = {
			text = has_open_minded_custom
			OR = {
				AND = {
					exists = dynasty
					dynasty = {
						has_dynasty_perk = witchcraft_legacy_1
					}
				}
				any_vassal = {
					exists = dynasty
					dynasty = {
						has_dynasty_perk = witchcraft_legacy_1
					}
				}
				faith = {
					has_doctrine_parameter = witchcraft_accepted
				}
				any_courtier = {
					has_trait = archmage
				}
			}
		}
		
	}

	is_shown_character = {
		scope:employee = {
			is_imprisoned = no
			OR = {
				is_vassal_of = scope:liege
				is_courtier_of = scope:liege
			}
		}
	}

	valid_character = {
		scope:employee = {
			is_imprisoned = no
			NOT = { has_trait = witch_hunter }
			#trigger_if = {
			#	limit = { NOT = { has_court_position = mage_court_position } }
			#	court_position_does_not_already_have_a_job_trigger = yes # This positions is considered a job, and cannot be held at the same tims as another similar position
			#}
		}
	}

	revoke_cost = {
		prestige =  {
			value = major_court_position_prestige_revoke_cost

			# Set to 0 with Dynasty Perk Inner Circle
			if = {
				limit = {
					exists = dynasty
					dynasty = { has_dynasty_perk = fp2_coterie_legacy_1 }

					employs_court_position = mage_court_position
					any_court_position_holder = {
				        type = mage_court_position
						exists = dynasty
						house = prev.house
					}
				}
				multiply = 0
			}
		}
	}

	salary = {
		round = no
		gold = {
			value = medium_court_position_salary
			# Set to 0 with Dynasty Perk Inner Circle
			if = {
				limit = {
					scope:liege = {
						exists = dynasty
						dynasty = { has_dynasty_perk = fp2_coterie_legacy_1 }

						employs_court_position = mage_court_position
						any_court_position_holder = {
					        type = mage_court_position
							exists = dynasty
							house = prev.house
						}
					}
				}
				multiply = {
					value = 0.0
					desc = inner_circle_salary_mod
					format = "BASE_VALUE_FORMAT_DECIMALS_MINUS_POSITIVE"
				}
			}
		}
	}

	# These modifiers are employed based on what level the aptitude reaches (levels are defined as breakpoints) 
	scaling_employer_modifiers = {
		terrible = {
			court_grandeur_baseline_add = 1
		}
		poor = {
			monthly_prestige = 0.1
			court_grandeur_baseline_add = 1
		}
		average = {
			court_grandeur_baseline_add = 2
			monthly_prestige = 0.15
		}
		good = {
			court_grandeur_baseline_add = 3
			monthly_prestige = 0.2
		}
		excellent = {
			court_grandeur_baseline_add = 3
			monthly_prestige = 0.25
		}
	}

	modifier = {
		monthly_prestige = medium_court_position_prestige_salary
		monthly_magic_lifestyle_xp_gain_mult = 0.1
	}

	on_court_position_received = {
		basic_gained_court_position_effect = yes
	}
	on_court_position_revoked = { 
		basic_revoked_court_position_effect = yes
	}
	on_court_position_invalidated = {

	}
	# Scripted value to determine whether an AI will hire/fire a position or not. Hires over 0, fires under -50. Look in _court_positions.info for more extensive documentation!
	candidate_score = {
		value = 50
		add = scope:base_value
		add = court_position_candidate_score_base_value
		add = court_position_candidate_aptitude_value
		add = court_position_debt_considerations_value
		if = {
			limit = {
				scope:employee = {
					exists = dynasty
					dynasty = {
						has_dynasty_perk = witchcraft_legacy_1
					}
				}
			}
			add = 50
		}
		if = {
			limit = {
				scope:employee = {
					has_character_flag = revoked_cmage
				}
			}
			add = -800
		}
		if = {
			limit = {
				scope:liege = {
					OR = {
						has_focus = arcane_focus
						has_trait = true_witch
						has_trait = archmage
					}
				}
			}
			add = -800
		}
		if = {
			limit = {
				scope:liege = {
					trait_is_criminal_in_faith_trigger = { TRAIT = witch FAITH = this.faith GENDER_CHARACTER = this }
					OR = {
						has_trait = zealous
						has_trait = theologian
						is_theocratic_lessee = yes
					}
				}
			}
			add = -800
		}
	}
}
