is_cannibal_trigger = {
	OR = {
		has_trait = cannibal
		any_secret = {
			secret_type = secret_cannibal
		}
	}
}


likes_target_trigger = {
	opinion = {
		target = $CHARACTER$
		value > 10
	}
}
