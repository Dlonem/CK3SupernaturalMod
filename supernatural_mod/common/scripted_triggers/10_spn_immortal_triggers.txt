spn_knows_about_character_being_vampire_trigger = {
	OR = {
		$CHARACTER$ = { has_trait = spn_known_vampire }
		any_known_secret = {
			secret_owner = $CHARACTER$
			secret_type = spn_secret_vampire
		}
	}
}
