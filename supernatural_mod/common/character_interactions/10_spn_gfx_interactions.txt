spn_disable_muscle_display = {
	category = interaction_category_friendly
	common_interaction = no

	desc = spn_disable_muscle_display_interaction_desc

	is_shown = {
		scope:recipient = {
			NOT = {
				has_character_flag = spn_no_muscle_display
			}
		}
		has_game_rule = spn_yes_right_click_visuals
	}

	on_accept = {
		scope:recipient = {
			add_character_flag = {
				flag = spn_no_muscle_display
			}

			show_as_tooltip = {
				scope:recipient = {
					custom_tooltip = spn_msg_tooltip_disable_muscle_display
				}
			}
		}

		save_scope_as = target
	}

	auto_accept = yes

	ai_will_do = {
		base = 0
	}
}

spn_assign_visual_age = {
	category = interaction_category_friendly
	common_interaction = no

	desc = spn_assign_visual_age_interaction_desc

	is_shown = {
		scope:recipient = {
			OR = {
				has_trait = spn_vampire
				has_trait = spn_hybrid
			}
		}
		has_game_rule = spn_yes_right_click_visuals
	}

	on_accept = {
		scope:recipient = {
			if = { 
				limit = { Is_AI = yes }
				scope:actor = {
					trigger_event = spn_utility_events.0001 #Trigger Age Setting Event
				}
			}
			else = {
				trigger_event = spn_utility_events.0001 #Trigger Age Setting Event
			}

			show_as_tooltip = {
				scope:recipient = {
					custom_tooltip = spn_msg_tooltip_assign_visual_age
				}
			}
		}

		save_scope_as = target
	}

	auto_accept = yes

	ai_will_do = {
		base = 0
	}
}
