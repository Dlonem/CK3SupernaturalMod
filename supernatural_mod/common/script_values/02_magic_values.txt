				has_character_flag = md_has_full_magic_prot_flag
			}
		}
		add = -3
	}

		limit = {
			OR = {
				AND = {
					has_trait = demon

					}

					has_trait = archmage

			}
		}

	}
	else_if = {
		limit = {

		}
