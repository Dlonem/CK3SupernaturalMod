# Known House Witches
activity_invite_rule_house_witches = {
	effect = {
		house ?= {
			every_house_member = {
				limit = {
					OR = {
						has_trait = witch
						has_trait = true_witch
						has_trait = grandmaster_coven
						has_trait = archmage
						has_trait = demon
						any_secret = {
							secret_type = secret_witch
							is_known_by = root
						}
					}
					# Remove diarchs from the list.
					bannable_serving_diarch_trigger = no
				}
				add_to_list = characters
			}
		}
	}
}
