

asp_witch_hunters = {
	#Can be both Teachers and Students
	#Only does progress since events is the primary way to gain friends and rivals during studies.
	icon = intrigue
	is_valid = {
		is_current_phase_active = yes
		any_attending_character = {
			OR = {
				has_character_flag = observed_by_whunters
				has_trait = paranoid
			}
		}
	}

	weight = {
		value = 1
	}

	effect = {
		random_attending_character = {
			save_scope_as = first
		}

		add_activity_log_entry = {
			key = asp_witch_hunters_key
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				add_stress = medium_stress_gain
			}
		}
	}
}

asp_procrastination = {
	#Either suitable trait or reveller intent
	icon = learning
	is_valid = {
		is_current_phase_active = yes
		any_attending_character = {
			OR = {
				has_trait = lazy
				has_trait = arrogant
				has_trait = drunkard
				has_trait = fickle
				has_trait = arbitrary
			}
		}
	}

	weight = {
		value = 1
	}

	effect = {
		random_attending_character = {
			limit = {
				OR = {
					has_trait = lazy
					has_trait = arrogant
					has_trait = drunkard
					has_trait = fickle
					has_trait = arbitrary
				}
			}
			save_scope_as = first
		}
		add_activity_log_entry = {
			key = apa_procrastination
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				add_stress = minor_stress_gain
			}
		}
	}
}

asp_night_on_books = {
	icon = learning
	is_valid = {
		is_current_phase_active = yes
	}

	weight = {
		value = 1
	}

	effect = {
		random_attending_character = {
			save_scope_as = first
		}

		add_activity_log_entry = {
			key = apa_night_on_books
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				add_stress = minor_stress_gain
				if = {
					limit = {
						this = scope:host
					}
					custom_tooltip = artifacts_searching_tt_positive_tiny
					scope:activity = { activity_special_type_progression_tiny = yes }
				}
			}
		}
	}
}
