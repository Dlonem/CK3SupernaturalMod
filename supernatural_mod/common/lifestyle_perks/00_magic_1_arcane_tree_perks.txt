magical_experiments_p1_perk = {
	lifestyle = magic_lifestyle
	tree = magic
	position = { 0.35 1 }
	icon = node_magic

	parent = protective_runes_perk

	auto_selection_weight = {
		value = 0
		if = {
			limit = {
				OR = {
					has_trait = witch
					any_secret = { secret_type = secret_witch }
				}
				OR = {
					any_character_artifact = { has_variable = magic1 }
					has_trait = source
				}
			}
			add = 2000
		}
	}

	can_be_picked = {
		has_focus=arcane_focus
	}
	
	effect = {
		
		random_list = {
			80 = {
				send_interface_toast = {
					title = magical_experiments_p1_perk_toast
					add_learning_skill = 1
				}
				
			}
			5 = {
				trigger = {
					NOT = {
						has_trait = disfigured
					}
				}
				add_trait = disfigured
			}
			5 = {
				trigger = {
					NOR = {
						has_trait = wounded_1
						has_trait = wounded_2
						has_trait = wounded_3
					}
				}
				add_trait = wounded_1
			}
			5 = {
				trigger = {
					NOT = {
						has_trait = one_eyed
					}
				}
				add_trait = one_eyed
			}
			5 = {
				trigger = {
					NOT = {
						has_trait = one_legged
					}
				}
				add_trait = one_legged
			}
		}
		if = { limit = { NOT = { has_character_modifier = power_saturated1 } }
			add_character_modifier = power_saturated1
		}
		custom_tooltip = magical_experiments_p1_perk_effect
		hidden_effect = {
			if = {
				limit = {
					is_ai = no
				}
				random = {
					chance = 80
					trigger_event = {
						id = magic_ev2.58
						years = { 2 15 }
					}
				}
			}
			random = {
				chance = 99
				trigger_event = {
					id = magic_ev2.59
					months = { 6 95 }
				}
			}
		}
	}
}
